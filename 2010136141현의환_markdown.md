### 파이썬 코드
~~~
import math 
 
def getPrime(number): #number가 소수이면 출력하는 함수
    if number == 2: #2는 소수이므로 바로 출력
        print number,
    for x in range(2,number): #2부터 number-1까지 나누어 본다
        if number % x == 0: #나누어 떨어지면 소수가 아니므로 break
            break
        elif (x > math.sqrt(number)): #sqrt(number)까지 나누어지지 않으면 소수
            print number,
            break
~~~
